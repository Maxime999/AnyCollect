84  size_t Metric::generateKey(const std::vector<std::string>& name, const std::map<std::string, std::string>& tags) {
85  std::string hash;
86  for (const auto& n : name)
87  hash.append(n);
88  for (const auto& [k, v] : tags) {
89  hash.append(k);
90  hash.append(v);
91  }
92  return Metric::hasher(hash);
93  }
